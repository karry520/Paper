{\huge 讲讲L2Norm2d是干什么的}
\begin{lstlisting}[caption={L2Norm2d}]
class L2Norm2d(nn.Module):
	def __init__(self, scale)：
		super(L2Norm2d, self).__init__()
		self.scale = scale
		
	def forward(self, x, dim = 1):
		return  self.scale * x *    \
				x.pow(2).sum(dim).  \
				clamp(min=1e-12).	\
				rsqrt().expand_as(x)
\end{lstlisting}
